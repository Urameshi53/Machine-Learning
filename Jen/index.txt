import matplotlib.pyplot as plt

def load(file):
    year = []
    month = []
    rain_temp = []
    with open(file,'r') as f:
        while True:
            t = f.readline()
            # break
            if not t:
                break
            t = t.split()
            # print(t)
            # break
            year.append(t[0])
            month.append(t[1])
            rain_temp.append(t[2])
    f.close()
    # print(year)
    # print(month)
    # print(rain_temp)
    # print(len(year))
    return year,month,rain_temp

def calculate_monthly(year,month,values):
    n = 0
    monthly = []
    monthly_values = {}
    year_modified = []
    while n <= 600:
        a = n
        n += 12
        if n > 600:
            break
        s = 0
        for i in range(a,n):
            s += float(values[i])
        monthly.append(round(s,2))
        year_modified.append(year[a])
    for i in range(len(year_modified)):
        monthly_values[year_modified[i]] = monthly[i]
    return monthly_values  

def calculate_yearly(monthly):
    yearly = monthly.copy()
    for i in monthly:
        yearly[i] = round(yearly[i]/12,2)
    return yearly
